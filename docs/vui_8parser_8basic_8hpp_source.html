127 bool parse_preprocessor() noexcept

129 if (region_.empty()) return true;

131 CharT c{};

161 bool parse_object(CharT c) noexcept
162 {
163 string_type name{ c };

202 bool read_value(std::any& out) noexcept
203 {
204 bool is_integer = true, is_decimal = true;
205
206 CharT c{ skip_whitespace()};
207 string_type s{};
208 bool is_negative{ false };
209 while ((c != ')') && (!stream_.eof()))
210 {
211 if (!isdigit(c))
212 {
213 if (c == '"')
214 {
215 if (s.back() != '\\')
216 {
217 is_integer = is_decimal = false;
218 if (!read_string(s)) return false;
219 }
220 else
221 s.back() = '"';
222 c = '\0';
223 }
224 else if (!is_negative && c == '-') is_negative = true;
225 else
226 {
227 is_integer = false;
228 if (c != '.') is_decimal = false;
229 }
230 }
231 s += c;
232 stream_ >> c;
233 }
234 if (stream_.eof()) return false;
235 if (s.empty()) out = "";
236 else if (is_integer) out = std::stoi(s);
237 else if (is_decimal) out = std::stod(s);
238 else if (s.size() >= 4 && s[0] == 't' && s[1] == 'r' && s[2] == 'u' && s[3] == 'e') out = true;
239 else if (s.size() >= 5 && s[0] == 'f' && s[1] == 'a' && s[2] == 'l' && s[3] == 's' && s[4] == 'e') out = false;
240 else out = s;
241 return true;
242 }
