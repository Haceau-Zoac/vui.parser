129 bool parse_preprocessor() noexcept
130 {
131 if (region_.empty()) return true;
132
133 CharT c{};

135 while (!is_region)
136 {
137 if (!(stream_ >> c) || c != '#') return false;

139 for (auto const& reg : region_)
140 if (!(stream_ >> c) || c != reg)

145 conti = true;

147 }

149 is_region = true;

163 bool parse_object(CharT c) noexcept
164 {
165 string_type name{ c };
166 if (!read_to('{', name)) return false;
167 object_type obj;

205 {
206 bool is_integer = true, is_decimal = true;
207
208 CharT c{ skip_whitespace()};
209 string_type s{};
210 bool is_negative{ false };
211 while ((c != ')') && (!stream_.eof()))
212 {
213 if (!isdigit(c))
214 {
215 if (c == '"')
216 {
217 if (s.back() != '\\')
218 {
219 is_integer = is_decimal = false;
220 if (!read_string(s)) return false;
221 }
222 else
223 s.back() = '"';
224 c = '\0';
225 }
226 else if (!is_negative && c == '-') is_negative = true;
227 else
228 {
229 is_integer = false;
230 if (c != '.') is_decimal = false;
231 }
232 }
233 s += c;
234 stream_ >> c;
235 }
236 if (stream_.eof()) return false;
237 if (s.empty()) out = "";
238 else if (is_integer) out = std::stoi(s);
239 else if (is_decimal) out = std::stod(s);
240 else if (s.size() >= 4 && s[0] == 't' && s[1] == 'r' && s[2] == 'u' && s[3] == 'e') out = true;
241 else if (s.size() >= 5 && s[0] == 'f' && s[1] == 'a' && s[2] == 'l' && s[3] == 's' && s[4] == 'e') out = false;
242 else out = s;
243 return true;
244 }
